## Samples

<Canvas>
  <Story
    name="Example"
    args={{
      isOpen: false,
    }}
  >
    {(args) => {
      const [isOpen, setIsOpen] = React.useState(args.isOpen);
      return (
        <>
          <Button inline={true} onClick={() => setIsOpen(true)}>
            OPEN SNACKBAR
          </Button>
          <Snackbar
            {...args}
            isOpen={isOpen}
            color="default"
            anchorOrigin={{
              vertical: "top",
              horizontal: "left",
            }}
            onClose={() => setIsOpen(false)}
          >
            This is Snackbar(default color).
          </Snackbar>
          <Snackbar
            {...args}
            isOpen={isOpen}
            color="dark"
            anchorOrigin={{
              vertical: "top",
              horizontal: "right",
            }}
            onClose={() => setIsOpen(false)}
          >
            This is Snackbar(dark color).
          </Snackbar>
          <Snackbar
            {...args}
            isOpen={isOpen}
            color="warning"
            anchorOrigin={{
              vertical: "bottom",
              horizontal: "right",
            }}
            onClose={() => setIsOpen(false)}
          >
            This is Snackbar(warning color).
          </Snackbar>
        </>
      );
    }}
  </Story>
</Canvas>
