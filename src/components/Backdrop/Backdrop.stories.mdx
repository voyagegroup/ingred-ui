## Samples

<Canvas>
  <Story
    name="Example"
    args={{
      isOpen: false,
    }}
  >
    {(args) => {
      const [isOpen, setIsOpen] = React.useState(false);
      const handleToggle = () => {
        setIsOpen(!isOpen);
      };
      return (
        <>
          <Button onClick={handleToggle}>Toggle Show Backdrop & Spinner</Button>
          <Backdrop
            {...args}
            isOpen={isOpen || args.isOpen}
            onClick={handleToggle}
          >
            <Spinner />
          </Backdrop>
        </>
      );
    }}
  </Story>
</Canvas>
